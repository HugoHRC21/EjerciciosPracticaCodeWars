#include <string>
#include <unordered_set>
#include <cctype>
using namespace std;

bool is_isogram(string str) {
    unordered_set<char> unique_chars;
    
    for (char c : str) {
        c = tolower(c);
        if (unique_chars.count(c) > 0) {
            return false;
        }
        unique_chars.insert(c);
    }
    
    return true;
}
